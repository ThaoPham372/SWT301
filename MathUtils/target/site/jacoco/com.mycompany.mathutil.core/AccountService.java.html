package com.mycompany.mathutil.core;

import java.util.Objects;
import java.util.regex.Pattern;

/**
 * Service đơn giản mô phỏng chức năng đăng ký tài khoản.
 *
 * Yêu cầu từ lab:
 * - password phải lớn hơn 6 ký tự
 * - email hợp lệ
 * - username không được rỗng
 */
public class AccountService {

    // Regex đơn giản kiểm tra định dạng email hợp lệ ở mức cơ bản
    private static final Pattern EMAIL_PATTERN =
            Pattern.compile("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");

    /**
     * Đăng ký tài khoản.
     *
     * @param username tên đăng nhập
     * @param password mật khẩu
     * @param email    địa chỉ email
     * @return true nếu dữ liệu hợp lệ theo rule của bài lab, ngược lại false
     */
    public boolean registerAccount(String username, String password, String email) {
        if (!isValidUsername(username)) {
            return false;
        }
        if (!isValidPassword(password)) {
            return false;
        }
        if (!isValidEmail(email)) {
            return false;
        }
        // Ở đây giả sử luôn đăng ký thành công nếu dữ liệu hợp lệ
        // (không kết nối DB hay lưu trạng thái thật)
        return true;
    }

    private boolean isValidUsername(String username) {
        return username != null && !username.trim().isEmpty();
    }

    public boolean isValidPassword(String password) {
        return password != null && password.length() > 6;
    }

    public boolean isValidEmail(String email) {
        if (Objects.isNull(email)) {
            return false;
        }
        return EMAIL_PATTERN.matcher(email).matches();
    }
}
